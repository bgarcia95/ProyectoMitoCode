--CATEGORIA--
CREATE TABLE categoria(id_categoria NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, 
nombre VARCHAR(50), estado number (1));


--PERSONA--
CREATE TABLE persona(id_persona NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, nombres VARCHAR(50),apellidos VARCHAR (50), sexo VARCHAR(1), fechaNacimiento DATE);


--TELEFONO--
CREATE TABLE telefono(id_telefono NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, id_persona number, numero VARCHAR(9),
FOREIGN KEY (id_persona) REFERENCES persona(id_persona));

--USUARIO--
CREATE TABLE usuario(id_usuario number PRIMARY KEY, usuario VARCHAR(20),clave VARCHAR(40),tipo VARCHAR(1),estado NUMBER (1),
FOREIGN KEY (id_usuario) REFERENCES persona(id_persona));


--NOTA--
CREATE TABLE nota(id_nota NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, id_persona number, 
id_categoria number, encabezado VARCHAR(50), 
cuerpo VARCHAR(500),fecha DATE,comentarioAdmin VARCHAR(50),valorizacion number(1), 
FOREIGN KEY (id_persona) REFERENCES persona(id_persona),
FOREIGN KEY (id_categoria) REFERENCES categoria(id_categoria));
